## 6. 실행 기록

```bash
# schema probe
PYTHONPATH=src python -m ais_risk.schema_probe_cli \
  --input data/raw/noaa/noaa_us_coastal_all_2023-08-01_2023-08-01_v1/raw.csv \
  --output outputs/noaa_us_coastal_all_2023-08-01_2023-08-01_v1_probe.json

# preprocess
PYTHONPATH=src python -m ais_risk.preprocess_cli \
  --input data/raw/noaa/noaa_us_coastal_all_2023-08-01_2023-08-01_v1/raw.csv \
  --output data/curated/noaa_us_coastal_all_2023-08-01_2023-08-01_v1_curated.csv

# trajectory
PYTHONPATH=src python -m ais_risk.trajectory_cli \
  --input data/curated/noaa_us_coastal_all_2023-08-01_2023-08-01_v1_curated.csv \
  --output data/curated/noaa_us_coastal_all_2023-08-01_2023-08-01_v1_tracks.csv

# workflow
PYTHONPATH=src python -m ais_risk.workflow_cli \
  --input data/raw/noaa/noaa_us_coastal_all_2023-08-01_2023-08-01_v1/raw.csv \
  --config configs/base.toml \
  --output-dir outputs/noaa_us_coastal_all_2023-08-01_2023-08-01_v1_workflow
```
